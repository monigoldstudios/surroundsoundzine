      html, body { margin: 0; padding: 0; overflow: hidden; }


      /* Floating iframe overlay */

      #webOverlay {

        position: absolute;

        width: 250px;      /* width of your zine page overlay */

        height: 140px;     /* height of your zine page overlay */

        top: 50%;

        left: 50%;

        transform: translate(-50%, -50%);

        display: none;     /* hidden until target found */

  <body>

    <!-- AR Scene -->

    <a-scene mindar-image="imageTargetSrc: https://cdn.jsdelivr.net/gh/monigoldstudios/rykardtest@main/rykardtest.mind;"

             color-space="sRGB"

             renderer="colorManagement: true, physicallyCorrectLights"

             vr-mode-ui="enabled: false"

             device-orientation-permission-ui="enabled: false"

             mindar-image-autostart="true">


      <a-camera position="0 0 0" look-controls="enabled: false"></a-camera>


      <a-entity mindar-image-target="targetIndex: 0" id="pageTarget">

        <!-- Optional placeholder plane -->

        <a-plane position="0 0 0" width="1" height="0.552" color="#fff"></a-plane>

      </a-entity>


    </a-scene>


    <!-- YouTube overlay -->

    <iframe id="webOverlay"

            src="https://www.youtube.com/embed/vNlKFm82aVU"

            allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture"

            allowfullscreen>

    </iframe>


    <script>

      const overlay = document.getElementById('webOverlay');

      const pageTarget = document.getElementById('pageTarget');

      const camera = document.querySelector('a-camera');


      pageTarget.addEventListener('targetFound', () => {

        overlay.style.display = 'block';

      });


      pageTarget.addEventListener('targetLost', () => {

        overlay.style.display = 'none';

      });


      // Function to update iframe position & rotation to simulate perspective

      function updateOverlay() {

        if (overlay.style.display === 'block') {

          const targetObj = pageTarget.object3D;

          const cameraObj = camera.object3D;


          // Project 3D position of target to 2D screen coordinates

          const vector = new THREE.Vector3();

          vector.setFromMatrixPosition(targetObj.matrixWorld);

          vector.project(cameraObj);


          // Convert normalized device coords to screen coords

          const x = (vector.x + 1) / 2 * window.innerWidth;

          const y = (-vector.y + 1) / 2 * window.innerHeight;


          overlay.style.left = `${x}px`;

          overlay.style.top = `${y}px`;


          // Optional: scale based on distance from camera

          const distance = cameraObj.position.distanceTo(targetObj.position);

          const scale = 1 / distance * 2; // tweak this factor for size

          overlay.style.width = `${250 * scale}px`;

          overlay.style.height = `${140 * scale}px`;


          // Optional: rotate iframe to roughly match marker rotation

          const rotY = THREE.Math.radToDeg(targetObj.rotation.y);

          overlay.style.transform = `translate(-50%, -50%) rotateY(${rotY}deg)`;

        }

        requestAnimationFrame(updateOverlay);

      }

      updateOverlay();

    </script>

  </body>
